626（Div.2）Codeforces Round

galagala · 2020-3-8 · 次阅读

前言

没啥说的，没思路的水题

题目链接：https://codeforces.ml/contest/1323/problem/B

B

题目大意：

给出n,m,k分别为矩阵的行数、列数、所要求的和
求出不包含零且子矩阵面积为k的数量。

Examples：

  • input

    3 3 2
    1 0 1
    1 1 1

  • output

    4

  • input

    3 5 4
    1 1 1
    1 1 1 1 1

  • output

    14

思路：

分解k，求出k的所有因子，分别求出n、m中连续1的长度包含的k的因子。
最终两结果相乘之和即为答案。
可能说的不是很详细，还是看代码吧，代码容易懂一些。

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int INF = 0x3f3f3f3f;
const int Maxn = 1e6+7;
#define pb push_back
#define me(a,b) memset(a,b,sizeof(a))
int a[Maxn],b[Maxn];
inline void Main()
{
    int n,m,k;
    cin>>n>>m>>k;
    for(int i=1; i<=n; i++) cin>>a[i];
    for(int i=1; i<=m; i++) cin>>b[i];
    vector<ll> v;
    for(int i=1; i*i<=k; i++)//求出k所有因数 
    {
        if(i*i==k)
        {
            v.pb(i);
            break;
        }
        else if(k%i==0)
        {
            v.pb(i),v.pb(k/i);
        }
    }
    sort(v.begin(),v.end());
    map<int,ll>ma,mb;
    int l=-1,r=-1;
    for(int i=1; i<=n; i++)//行 
    {
        if(a[i]&&l==-1) l=i;
        if(!a[i]&&l!=-1) r=i-1;
        if(a[i]&&i==n) r=i;
        int p=r-l+1;    //求出每段连续1的长度 
//        cout<<p<<endl;
        if(r!=-1)
        {
            for(auto it:v)
            {
                if(it>p) break;
                ma[it] += p-it+1;//这里有点晕 大致求的就是取该因数时能凑出来的份数 
            }
            l=-1,r=-1;
        }
    }
    for(int i=1; i<=m; i++)//列 
    {
        if(b[i]&&l==-1) l=i;
        if(!b[i]&&l!=-1) r=i-1;
        if(b[i]&&i==m) r=i;
        int p=r-l+1;
//        cout<<p<<endl;
        if(r!=-1)
        {
            for(auto it:v)
            {
                if(it>p) break;
                mb[it] += p-it+1;
            }
            l=-1,r=-1;
        }
    }
    ll ans=0;
    for(auto it:v)
    {
        ans+=ma[it]*mb[k/it];//乘积之和即为结果 
    }
    cout<<ans<<endl;
}

int main()
{
    Main();
    return 0;
}
